create collection 

        
            import pouchlite from '@pouchlabs/lite';
            import {object,string} from '@pouchlabs/lite'
            //init
            let db = pouchlite().init();
            
            //schema
             let schema = object({
                name:string(),
                title:string(),
                text:string()
             })
            //create collection
            db.useDb('blog').createCollection('comments',schema);

remove collection

        
            db.useDb('blog').removeColection('comments')
            //danger removes data and the collection itself
